var time = 0;                               //变量？

var timer = setInterval(function() {
  time += 2;
  console.log(time + "seconds have passed");//time自动转化成字符串
  if (time > 5) {
    clearInterval(timer);                   //终止timer计时器
  }
}, 2000);
